Пример:
```C#
// Класс содержащий дополнительную информацию о событии
// Обязан наследоваться от EventArgs
public class InformationEventArgs(int value) : EventArgs
{
	public int Value { get; } = value;
}

// Класс который владеет событием и через который оно вызывается 
public class Alarm
{
	// Поле события, можно подписаться и отписаться
	public event EventHandler<InformationEventArgs>? Ring;
	
	// Метод, который делает что-то и как побочный эффект вызывает событие
	public void Trigger(int value)
	{
		Ring?.Invoke(this, new InformationEventArgs(value));
	}
}

// Класс который прописал Обработчик событий и может подписать его на событие
public class Subscriber
{
	// Название обработчика обычно начинается с On
	// Сигнатура соответствует EventHandler<T>
	public void OnAlarmRing(object? sender, InformationEventArgs e)
	{
		Console.Writeline("Riing!");
	}
}

class Program
{
	static void Main()
	{
		// Создали объект, который вызывает Событие
		Alarm testAlarm = new();
		Subscriber testSubscriber = new();
		
		// Подписали обработчик события на событие
		testAlarm.Ring += testSubscriber.OnAlarmRing;
		
		// Вызвали событие
		testAlarm.Trigger();  // Выведет Riing!
	}
}
```
