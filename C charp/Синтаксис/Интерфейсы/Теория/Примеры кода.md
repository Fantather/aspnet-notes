Пример с реализациями по умолчанию
```C#
interface IMyCollection
{
    int Count { get; }

    int GetItem(int index);   // метод получения элемента
    void SetItem(int index, int value); // метод установки элемента

    // индексатор с реализацией по умолчанию
    int this[int index]
    {
        get => GetItem(index);
        set => SetItem(index, value);
    }
}
```

Пример использования
```C#
// Объявляем интерфейс
interface IDrawable
{
    void Draw(); // Контракт: "Все, что можно нарисовать, должно иметь метод Draw"
}

// Класс реализует интерфейс
class Circle : IDrawable
{
    public void Draw()
    {
        Console.WriteLine("Рисую круг");
        // Здесь находится реальная реализация метода
    }
}

class Square : IDrawable
{
    public void Draw()
    {
        Console.WriteLine("Рисую квадрат");
        // Своя собственная реализация для квадрата
    }
}

// Использование
class Program
{
    static void Main()
    {
        // Мы можем обращаться к разным объектам через общий интерфейс
        List<IDrawable> shapes = new List<IDrawable> { new Circle(), new Square() };

        foreach (var shape in shapes)
        {
            shape.Draw(); // Вызовется правильная реализация для каждого объекта
        }
    }
}
```
