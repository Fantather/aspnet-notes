Деконструктор объектов - Синтаксический сахар, метод, который позволяет разбирать объект на части
Его используют для удобного присваивания значений из объекта в отдельные переменные

Пример деконтруктора
```C#
var person = new Person("Alice", 25, 1.70);

// Создали 3 переменных с разными типами
var (name, age, height) = person;

Console.WriteLine($"{name}, {age}, {height}");


class Person
{
    public string Name { get; }
    public int Age { get; }
    public double Height { get; }

    public Person(string name, int age, double height)
    {
        Name = name;
        Age = age;
        Height = height;
    }

    // деконструктор
    public void Deconstruct(out string name, out int age, out double height)
    {
        name = Name;
        age = Age;
        height = Height;
    }
}
```
