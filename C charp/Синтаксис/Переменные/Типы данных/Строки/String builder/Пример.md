```csharp
var sb = new StringBuilder(50);   // заранее резервируем место
sb.Append("ID: ");
sb.Append(123);
sb.Append(", Name: ");
sb.AppendLine("Alice");

// Вставим дату в начало
sb.Insert(0, DateTime.Now.ToString("yyyy-MM-dd") + " — ");

// Заменим запятую на точку с запятой
sb.Replace(',', ';');

// Получим готовую строку
string result = sb.ToString();
Console.WriteLine(result);
```
