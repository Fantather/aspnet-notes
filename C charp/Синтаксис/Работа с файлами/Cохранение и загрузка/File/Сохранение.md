```C#
using System;
using System.IO;
using System.Text.Json;

// Предположим, у нас уже есть jsonString
string jsonString = "{\"Name\":\"Alice\",\"Age\":30,\"Birthday\":\"1993-05-10T00:00:00\"}";

// Указываем путь к файлу
string filePath = "userdata.json";

// Записываем строку в файл (перезаписывает файл, если он существует)
File.WriteAllText(filePath, jsonString);

Console.WriteLine($"JSON успешно сохранен в файл: {Path.GetFullPath(filePath)}");
```
