`JsonConverter` - Позволяет использовать пользовательский конвертер для свойств или типов
Это самый мощный инструмент для обработки сложных сценариев

**Пример**
Конвертер для сложного типа `Address`
```C#
public class Address
{
    public string Street { get; set; }
    public string City { get; set; }
}

public class AddressConverter : JsonConverter<Adress>
{
	public override Address Read(ref UtfJsonReader reader, Type typeToConvert, JsonSerializerOptions options)
	{
		// Читаем строку и парсим ее в объект Address
		var adressString = reader.GetString();
		var parts = addressString?.Split(',');
		return new Address
		{
			Street = parts?[0] ?? "",
			City = parts?[1] ?? ""
		};
	}
	
	public override void Write(UtfJsonWriter writer, Address value, JsonSerializerOptions options)
	{
		// Записываем объект Address как единую строку
		writer.WriteStringValue($"{value.Street}, {value.City}");
	}
}
```

**Использование**
```C#
public class Company
{
    public string Name { get; set; }
    
    [JsonConverter(typeof(AddressConverter))]
    public Address OfficeAddress { get; set; }
}

// Main
var company = new Company {
    Name = "Tech Corp",
    OfficeAddress = new Address { Street = "123 Main St", City = "London" }
};
var json = JsonSerializer.Serialize(company);
// Результат: {"Name":"Tech Corp","OfficeAddress":"123 Main St,London"}
```
