По умолчанию `System.Text.Json` сериализует `DateTime` и `DateTimeOffset` в формат по стандарту ISO 8601 (например, `"2023-10-25T14:30:45.1234567Z"`)
Но лучше настраивать формат времени глобально, через `JsonSerializerOptions`

```C#
public class Something
{
    [JsonConverter(typeof(DateFormatConverter))]
    public DateTime Date { get; set; }
}

public class DateFormatConverter : JsonConverter<DateTime>
{
    public override DateTime Read(ref Utf8JsonReader reader, Type typeToConvert, JsonSerializerOptions options)
        => DateTime.ParseExact(reader.GetString(), "yyyy-MM-dd", null);

    public override void Write(Utf8JsonWriter writer, DateTime value, JsonSerializerOptions options)
        => writer.WriteStringValue(value.ToString("yyyy-MM-dd"));
}

// Использование:
var something = new Something { Date = DateTime.Now };
var json = JsonSerializer.Serialize(something);
// Результат: {"Date":"2023-10-25"}
```
