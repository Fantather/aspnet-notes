Создание класса
Класс не требует специальных атрибутов, но они дают больше контроля
```C#
public class Order
{
	[XmlAttribute] // Сделает Id атрибутом тега <Order>
	public int Id { get; set; }
	
	[XmlElement("ProductName")] // Изменит имя элемента в XML
	public string Name { get; set; }
	
	public decimal Price { get; set; }
}
```


```C#
Order order = new { Id = 1, Name = "Laptop", Price = 999.99m };
var serializer = new XmlSerializer(typeof(Order));

// Сериализация в файл
using (var writer = new StreamWriter("order.xml"))
{
	serializer.Serialize(writer, order);
}

// Десериализация из файла
using (var reader = new StremReader("order.xml"))
{
	Order deserializedOrder = (Order)serializer.Deserialize(reader);
}
```
