Согласованность - процесс в ходе которого реакт сравнивает виртуального древо

цель - оптимизировать производительность

```jsx
import React, { useState } from 'react';

function App() {

  const [count, setCount] = useState(0);

  const handleIncrement = () => {

    setCount(count + 1);

  };

  return (

    <div>

      <h2>Counter: {count}</h2>

      <button onClick={handleIncrement}>Increment</button>

    </div>

  );

};

export default App;
```

что тут происходит
пользователь нажимает кнопку инкремент несколько раз
при каждом нажатии обновляется состояние счетчика
запускается согласование для обновление юзер интерфейса

там мы обновили значение нашей переменной
