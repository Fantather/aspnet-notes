что бы создать свой классовый компонент
мы должны расширить базовый компонент

```jsx
import React from "react";

function App() {

  return (

    <div>

      {" "}

      <h1>Firt component</h1>

      <MyComponent />

    </div>

  );

}


class MyComponent extends React.Component {

  render() {

    return "Im func component";

  }

}

  

export default App;
```
