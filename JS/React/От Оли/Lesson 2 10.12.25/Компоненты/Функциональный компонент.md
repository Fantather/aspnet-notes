в самозакрывающихся тегах необходима закрывающаяся черта(не составные)
![[Pasted image 20251210182955.png]]
```jsx
function App() {

  return (

    <div>

      {" "}

      <h1>Firt component</h1>

      <MyComponent />

    </div>

  );

}


function MyComponent() {

  return "Im func component";

}


export default App;
```

оба пропса доступны только для чтения
они доступны для передачи настроек от родительских компонентов к дочерним
они не универсальное хранилище данных
