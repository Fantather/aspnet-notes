именованный экспорт
если мы хотим экспортировать несколько ф-ций и компонентов по умолчанию

```jsx
/*Lesson*/

import React from 'react';

import {Footer, Header } from './components/exampleComponents/myComponents.jsx';

  

class App extends React.Component {

  titleText = "Hello world!";

  helpText = 'Help text.';

  render() {

    return (

      <div>

        <Header title="Header Text" />

        <h1>{this.titleText}</h1>

        <input placeholder={this.helpText} onClick={this.inputClick} onMouseEnter={this.mouseOver}></input>

        <Footer />

      </div>);

  

  }

  inputClick = () => console.log('Clicked!');

  mouseOver = () => console.log('Mouse over!');

}

export default App;
```


```jsx 
export const Header = () => {

    return <h1>Hello, world!</h1>;

};

export const Footer = () => {

    return <footer>Goodbye, world!</footer>;

};
```
