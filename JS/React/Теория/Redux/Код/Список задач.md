src/reducers/todoReducer.js
```js
const initialState = {
  todos: [],
  filter: "all",
};

export function todoReducer(state = initialState, action) {
  switch (action.type) {
    case "ADD_TODO":
      return {
        ...state,
        todos: [
          ...state.todos,
          {
            id: action.payload.id,
            text: action.payload.text,
            completed: false,
          },
        ],
      };

    case "REMOVE_TODO":
      return {
        ...state,
        todos: state.todos.filter((t) => t.id !== action.payload),
      };

    case "TOGGLE_TODO":
      return {
        ...state,
        todos: state.todos.map((t) =>
          t.id === action.payload ? { ...t, completed: !t.completed } : t
        ),
      };

    case "SET_FILTER":
      return {
        ...state,
        filter: action.payload,
      };

    case "CLEAR_TASKS":
      return {
        ...state,
        todos: [],
      };

    default:
      return state;
  }
}
```

src/actions/todoActions.js
```js
import { store } from "../store";

export const todoActions = {
  addTodo(text) {
    store.dispatch({
      type: "ADD_TODO",
      payload: {
        id: Date.now(),
        text,
      },
    });
  },

  removeTodo(id) {
    store.dispatch({
      type: "REMOVE_TODO",
      payload: id,
    });
  },

  toggleTodo(id) {
    store.dispatch({
      type: "TOGGLE_TODO",
      payload: id,
    });
  },

  setFilter(filter) {
    store.dispatch({
      type: "SET_FILTER",
      payload: filter,
    });
  },

  clearTasks() {
    store.dispatch({
      type: "CLEAR_TASKS",
    });
  },
};
```


App.jsx
```jsx
import { useEffect, useState } from "react";
import "./App.css";

// предполагаем, что store и todoActions уже созданы и импортируются
import { store } from "./store";
import { todoActions } from "./actions/todoActions";

export default function App() {
  // локальный state = копия redux state для перерендера
  const [state, setState] = useState(store.getState());
  const [input, setInput] = useState("");

  // подписка на Redux store (аналог store.subscribe)
  useEffect(() => {
    const unsubscribe = store.subscribe(() => {
      setState(store.getState());
    });
    return unsubscribe; // отписка при размонтировании
  }, []);

  const { todos, filter } = state;

  const visibleTodos = todos.filter((todo) => {
    if (filter === "all") return true;
    if (filter === "active") return !todo.completed;
    if (filter === "completed") return todo.completed;
    return true;
  });

  function handleAdd() {
    const text = input.trim();
    if (!text) return;
    todoActions.addTodo(text);
    setInput("");
  }

  function handleKeyDown(e) {
    if (e.key === "Enter") handleAdd();
  }

  return (
    <div className="app">
      <h1>ToDo (Redux)</h1>
      
      <div className="form-row">
        <input
          type="text"
          placeholder="A new task..."
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyDown={handleKeyDown}
        />
        <button onClick={handleAdd}>Add</button>
      </div>
      
      <div className="filters small">
        <button onClick={() => todoActions.setFilter("all")}>All</button>
        <button onClick={() => todoActions.setFilter("active")}>Active</button>
        <button onClick={() => todoActions.setFilter("completed")}>
          Completed
        </button>
        <button
          onClick={() => {
            if (confirm("Clear all tasks?")) {
              todoActions.clearTasks();
            }
          }}
        >
          Clear
        </button>
      </div>
      
      <ul className="todo-list">
        {visibleTodos.map((todo) => (
          <li key={todo.id} className="todo-item">
            <div>
              <input
                type="checkbox"
                checked={todo.completed}
                onChange={() => todoActions.toggleTodo(todo.id)}
              />
              <span
                className={`todo-text ${todo.completed ? "completed" : ""}`}
              >
                {todo.text}
              </span>
            </div>
            <div>
              <button onClick={() => todoActions.removeTodo(todo.id)}>
                Delete
              </button>
            </div>
          </li>
        ))}
      </ul>
    </div>
  );
}
```
