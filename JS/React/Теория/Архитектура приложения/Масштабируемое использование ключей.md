```JSX
const THEMES = {
  LIGHT: 'light',
  DARK: 'dark'
}

const themesMap = {
  [THEMES.LIGHT]: {
    box: 'card white-theme',
    btn: 'white-theme button',
    btnText: 'Сменить на тёмную тему',
    nextTheme: [THEMES.DARK]
  },

  [THEMES.DARK]: {
    box: 'card dark-theme',
    btn: 'dark-theme button',
    btnText: 'Сменить на светлую тему',
    nextTheme: [THEMES.LIGHT]
  }
}


function EditingTheme()
{
  const [theme, setTheme] = useState('light');
  const {box, btn, btnText, nextTheme} = themesMap[theme];
  return(
    <>
      <div className={box}>
        <button onClick={()=>{setTheme(nextTheme)}} className={btn}>{btnText}</button>
      </div>
    </>
  )
}
```
