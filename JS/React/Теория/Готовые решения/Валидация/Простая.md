function isLoginValid(login)
{
    const minLength = 3;
    const maxLength = 20;
    const loginRegex = /^[a-zA-Z0-9_]+$/;

    return (
        login.length >= minLength &&
        login.length <= maxLength &&
        loginRegex.test(login)
    );
}
