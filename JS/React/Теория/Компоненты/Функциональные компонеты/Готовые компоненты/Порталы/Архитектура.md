Порталы хранятся в `src/portals`
Внутри можно создать файл на пример `portalLayers.js` и в нём словарь `PORTAL_LAYERS`
```JS
export const PORTAL_LAYERS = {
  tooltips: document.getElementById("portal-tooltips"),
  modals: document.getElementById("portal-modals"),
  notifications: document.getElementById("portal-notifications"),
  system: document.getElementById("portal-system"),
};
```

##### Компонент для обращения к существующим порталам (Или нет)
```JSx
import { createPortal } from "react-dom";

export function Portal({ container, children }) {
  if (!container) return null;
  return createPortal(children, container);
}
```


Использование этой функции
Код компонента Подсказки
```JSX
import { Portal } from "../portals/Portal";
import { PORTAL_LAYERS } from "../portals/portalLayers";

export function Tooltip({ text, x, y }) {
  return (
    <Portal container={PORTAL_LAYERS.tooltips}>
      <div
        style={{
          position: "fixed",
          top: y,
          left: x,
          padding: "4px 8px",
          background: "black",
          color: "white",
          borderRadius: 4,
          fontSize: 12,
        }}
      >
        {text}
      </div>
    </Portal>
  );
}
```

Код компонента Модального окна
```JSX
import { Portal } from "../portals/Portal";
import { PORTAL_LAYERS } from "../portals/portalLayers";

export function Modal({ onClose, children }) {
  return (
    <Portal container={PORTAL_LAYERS.modals}>
      <div
        onClick={onClose}
        style={{
          position: "fixed",
          inset: 0,
          background: "rgba(0,0,0,0.4)",
          display: "grid",
          placeItems: "center",
        }}
      >
        <div
          onClick={(e) => e.stopPropagation()}
          style={{
            background: "white",
            padding: 20,
            width: 400,
          }}
        >
          {children}
        </div>
      </div>
    </Portal>
  );
}
```

App.jsx
```JSX
import { useState } from "react";
import { Tooltip } from "./components/Tooltip";
import { Modal } from "./components/Modal";
import { GlobalLoader } from "./components/GlobalLoader";
import { Notification } from "./components/Notification";
function App() {
  const [showModal, setShowModal] = useState(false);
  const [showLoader, setShowLoader] = useState(false);
  return (
    <>
      <button
        onMouseEnter={() => {}}
        onClick={() => setShowModal(true)}
        style={{ margin: 40 }}
      >
        Open modal
      </button>
      <Tooltip text="I am tooltip" x={100} y={100} />
      {showModal && (
        <Modal onClose={() => setShowModal(false)}>Modal content</Modal>
      )}
      <Notification />
      {showLoader && <GlobalLoader />}
    </>
  );
}

export default App;
```
