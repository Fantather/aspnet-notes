App.jsx
```JSX
import React from "react";
import ReactDOM from "react-dom";

const App = () => {
  const [isOpen, setIsOpen] = React.useState(false);
  return (
    <div className="layout" style={{ overflow: "hidden" }}>
      <h1>Portal Example</h1>
      <button onClick={() => setIsOpen(true)}>Open the modal</button>
      {/* Логически Modal находится здесь, внутри div с overflow: hidden.
         Но физически в HTML он окажется в div#portal-root.
      */}
      {isOpen && (
        <Modal onClose={() => setIsOpen(false)}>
          <h2>I'm in the portal!</h2>
          <p>Parent styles (overflow: hidden) do not affect me.</p>
        </Modal>
      )}
    </div>
  );
};

export default App;

// Компонент самого модального окна
const Modal = ({ children, onClose }) => {
  // 1. Находим элемент, куда будем "телепортировать"
  const portalRoot = document.getElementById("portal-root");
  if (!portalRoot) return null;
  // 2. Возвращаем портал
  return ReactDOM.createPortal(
    <div className="modal-overlay">
      <div className="modal-content">
        <button onClick={onClose}>Close</button>
        {children}
      </div>
    </div>,
    portalRoot // <-- Цель назначения
  );
};
```
