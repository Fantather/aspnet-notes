```JSX
import './App.css'
import React from 'react';

  
function App() {

  return (
    <>
      <Table />
    </>
  )
}


const Columns = () => {
  return (
    <>
      <td>Cell 1</td>
      <td>Cell 2</td>
    </>
  );
};

const Table = () => {
  return (
    <table>
      <tbody>
        <tr>
          <Columns />
          <GlossaryList />
        </tr>
      </tbody>
    </table>
  );
};

  
const glossary = [
  { id: 1, term: "HTML", desc: "Markup language" },
  { id: 2, term: "CSS", desc: "Style tables" },
];

const GlossaryList = () => {
  return (
    <dl>
      {glossary.map((item) => (
        // Здесь обязательно React.Fragment, так как нам нужен key
        <React.Fragment key={item.id}>
          <dt>{item.term}</dt>
          <dd>{item.desc}</dd>
        </React.Fragment>
      ))}
    </dl>
  );
};


export default App
```
