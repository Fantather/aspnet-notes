CSSTransition - библиотека, которая даёт нам компонент которому можно настраивать поведение перед удалением

Установка `npm install react-transition-group`

```JSX
import React, { useState, useRef } from "react";
import ReactDOM from "react-dom";
import { CSSTransition } from "react-transition-group";
import "./assets/css/modal.css"; // стили для анимации

const portalRoot = document.getElementById("portal-root");

const Modal = ({ show, onClose, children }) => {
  const nodeRef = useRef(null);
  return ReactDOM.createPortal(
    <CSSTransition
      in={show}
      timeout={300}
      classNames="fade"
      unmountOnExit
      nodeRef={nodeRef} // важно
    >
      <div ref={nodeRef} className="modal-overlay" onClick={onClose}>
        <div className="modal-content" onClick={(e) => e.stopPropagation()}>
          <button onClick={onClose}>Close</button>
          {children}
        </div>
      </div>
    </CSSTransition>,
    portalRoot
  );
};

export default function App() {
  const [isOpen, setIsOpen] = useState(false);
  return (
    <div>
      <h1>Animated Portal Example</h1>
      <button onClick={() => setIsOpen(true)}>Open Modal</button>
      <Modal show={isOpen} onClose={() => setIsOpen(false)}>
        <p>This modal fades in and out!</p>
      </Modal>
    </div>
  );
}
```
