```jsx
import React, { useReducer, useCallback } from "react";

// --- 1. Логика Редьюсера ---
const initialCartState = {
  items: [],
  totalAmount: 0,
};

const cartReducer = (state, action) => {
  if (action.type === "ADD") {
    const updatedItems = state.items.concat(action.item);
    const updatedTotalAmount =
      state.totalAmount + action.item.price * action.item.amount;
    return {
      items: updatedItems,
      totalAmount: updatedTotalAmount,
    };
  }

  if (action.type === "REMOVE") {
    const existingItemIndex = state.items.findIndex(
      (item) => item.id === action.id
    );

    const existingItem = state.items[existingItemIndex];
    if (!existingItem) return state;
    
    let updatedItems;
    if (existingItem.amount === 1) {
      updatedItems = state.items.filter((item) => item.id !== action.id);
    } else {
      const updatedItem = { ...existingItem, amount: existingItem.amount - 1 };
      updatedItems = [...state.items];
      updatedItems[existingItemIndex] = updatedItem;
    }

    const updatedTotalAmount = state.totalAmount - existingItem.price;

    return {
      items: updatedItems,
      totalAmount: updatedTotalAmount,
    };
  }

  if (action.type === "CLEAR") {
    return initialCartState;
  }

  return state;
};

// --- 2. Создание Контекста ---

const CartContext = React.createContext({
  items: [],
  totalAmount: 0,
  addItem: (item) => {},
  removeItem: (id) => {},
  clearCart: () => {},
});

export const CartProvider = (props) => {
  const [cartState, dispatchCartAction] = useReducer(
    cartReducer,
    initialCartState
  );

  const addItemToCartHandler = useCallback((item) => {
    dispatchCartAction({ type: "ADD", item });
  }, []);

  const removeItemFromCartHandler = useCallback((id) => {
    dispatchCartAction({ type: "REMOVE", id });
  }, []);

  const clearCartHandler = useCallback(() => {
    dispatchCartAction({ type: "CLEAR" });
  }, []);

  const cartContext = {
    items: cartState.items,
    totalAmount: cartState.totalAmount,
    addItem: addItemToCartHandler,
    removeItem: removeItemFromCartHandler,
    clearCart: clearCartHandler,
  };

  return (
    <CartContext.Provider value={cartContext}>
      {props.children}
    </CartContext.Provider>
  );
};

export default CartContext;
```

CartButton.jsx
```jsx
import React, { useContext } from "react";
import CartContext from "./CartContext";

function CartButton() {
  const cartCtx = useContext(CartContext);
  const clearCartHandler = () => {
    cartCtx.clearCart();
  };

  const numberOfItems = cartCtx.items.reduce(
    (curNumber, item) => curNumber + item.amount,
    0
  );

  return (
    <>
      <button>
        <span>Cart</span>
        <span>{numberOfItems}</span>
      </button>
      <button onClick={clearCartHandler}>Clear cart</button>
    </>
  );
}

export default CartButton;
```

itemsList.jsx
```jsx
import React, { useContext } from "react";
import CartContext from "./CartContext";

const DUMMY_ITEMS = [
  { id: "i1", name: "Product 1", price: 10 },
  { id: "i2", name: "Product 2", price: 20 },
];

function ItemsList() {
  const cartCtx = useContext(CartContext);
  const addToCartHandler = (item) => {
    cartCtx.addItem({ ...item, amount: 1 });
  };

  return (
    <ul>
      {DUMMY_ITEMS.map((item) => (
        <li key={item.id}>
          {item.name} - ${item.price}
          <button onClick={() => addToCartHandler(item)}>Add to Cart</button>
        </li>
      ))}
    </ul>
  );
}

export default ItemsList;
```

```jsx
import React, { useContext } from "react";
import CartContext from "./CartContext";
import CartButton from "./CartButton";

function Header() {
  const cartCtx = useContext(CartContext);

  return (
    <header
      style={{
        padding: "1rem",
        backgroundColor: "#ccc",
        display: "flex",
        justifyContent: "space-between",
        alignItems: "center",
      }}
    >
      <h1>Shop</h1>
      <div style={{ display: "flex", gap: "1rem", alignItems: "center" }}>
        <CartButton />
        <span>Total: ${cartCtx.totalAmount.toFixed(2)}</span>
      </div>
    </header>
  );
}

export default Header;
```
