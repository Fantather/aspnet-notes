Все поля должны быть заполнены
```jsx
import React, { useReducer } from "react";

// 1. Редьюсер для формы
function formReducer(state, action) {
  switch (action.type) {
    case 'CHANGE_FIELD':
      return {
        ...state,

        [action.field]: action.value, // обновляем конкретное поле

      };

    case 'TOGGLE_VALIDITY':

      return {

        ...state,

        isValid: action.isValid,

      };

    default:

      return state;

  }

}

// 2. Начальное состояние формы

const initialFormState = {

  firstName: '',

  lastName: '',

  email: '',

  isValid: false

};

export default function App() {

  const [formData, dispatch] = useReducer(formReducer, initialFormState);

  // Простая проверка валидности: все поля должны быть заполнены

  const validateForm = () => {

    const isValid = formData.firstName && formData.lastName && formData.email;

    dispatch({ type: 'TOGGLE_VALIDITY', isValid });

  };

  return (

    <div style={{ padding: 20 }}>

      <h1>Form with useReducer</h1>

      <div style={{ marginBottom: 12 }}>

        <label>

          First Name: 

          <input

            type="text"

            value={formData.firstName}

            onChange={(e) => dispatch({ type: 'CHANGE_FIELD', field: 'firstName', value: e.target.value })}

            style={{ marginLeft: 8 }}

          />

        </label>

      </div>

      <div style={{ marginBottom: 12 }}>

        <label>

          Last Name: 

          <input

            type="text"

            value={formData.lastName}

            onChange={(e) => dispatch({ type: 'CHANGE_FIELD', field: 'lastName', value: e.target.value })}

            style={{ marginLeft: 8 }}

          />

        </label>

      </div>

      <div style={{ marginBottom: 12 }}>

        <label>

          Email: 

          <input

            type="email"

            value={formData.email}

            onChange={(e) => dispatch({ type: 'CHANGE_FIELD', field: 'email', value: e.target.value })}

            style={{ marginLeft: 8 }}

          />

        </label>

      </div>

      <button onClick={validateForm} style={{ padding: '6px 12px', marginTop: 10 }}>Validate Form</button>

      <p style={{ marginTop: 12 }}>

        Form is {formData.isValid ? 'valid' : 'invalid'}

      </p>

    </div>

  );

}
```
