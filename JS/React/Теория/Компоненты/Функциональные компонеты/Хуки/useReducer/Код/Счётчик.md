```jsx
import React, { useReducer } from "react";

export default function App() {
  // 1. Начальное состояние
  const initialState = { count: 0 };

  // 2. Редьюсер: принимает текущее состояние и действие, возвращает новое состояние
  function reducer(state, action) {
    switch (action.type) {
      case 'increment':
        return { count: state.count + 1 };
      case 'decrement':
        return { count: state.count - 1 };
      case 'reset':
        return initialState;
      default:
        return state;
    }
  }

  // 3. Используем useReducer
  const [state, dispatch] = useReducer(reducer, initialState);
  return (
    <div style={{ padding: 20 }}>
      <h1>useReducer Example</h1>
      <p>Current Count: {state.count}</p>
      <button onClick={() => dispatch({ type: 'increment' })} style={{ marginRight: 8 }}>Increment</button>
      <button onClick={() => dispatch({ type: 'decrement' })} style={{ marginRight: 8 }}>Decrement</button>
      <button onClick={() => dispatch({ type: 'reset' })}>Reset</button>
    </div>
  );
}
```
