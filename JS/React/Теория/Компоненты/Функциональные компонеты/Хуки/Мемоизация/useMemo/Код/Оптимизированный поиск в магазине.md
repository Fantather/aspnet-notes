```JSX
import { useState, useMemo } from "react";

const products = [
  { id: 1, name: "Laptop", price: 1200 },
  { id: 2, name: "Phone", price: 800 },
  { id: 3, name: "Tablet", price: 600 },
  { id: 4, name: "Monitor", price: 400 },
  { id: 5, name: "Keyboard", price: 150 },
];

function App() {
  return (
    <div>
      <ProductList />
    </div>
  );
}

export default App;
function ProductList() {
  const [search, setSearch] = useState("");
  const [sortByPrice, setSortByPrice] = useState(false);
  const [counter, setCounter] = useState(0);

  // useMemo для оптимизации фильтрации и сортировки

  const filteredProducts = useMemo(() => {
    console.log("Filtering and sorting products...");
    let result = products.filter((product) =>
      product.name.toLowerCase().includes(search.toLowerCase())
    );

    if (sortByPrice) {
      result = [...result].sort((a, b) => a.price - b.price);
    }

    return result;
  }, [search, sortByPrice]);

  return (
    <div>
      <h2>Product list</h2>
      <input
        type="text"
        placeholder="Search product"
        value={search}
        onChange={(e) => setSearch(e.target.value)}
      />
      <label>
        <input
          type="checkbox"
          checked={sortByPrice}
          onChange={() => setSortByPrice((prev) => !prev)}
        />
        Sort by price
      </label>
      <ul>
        {filteredProducts.map((product) => (
          <li key={product.id}>
            {product.name} - ${product.price}
          </li>
        ))}
      </ul>
      <hr />
      {/* Независимый стейт */}
      <button onClick={() => setCounter(counter + 1)}>
        Increase counter: {counter}
      </button>
      <p>Counter changes do not trigger filtering or sorting again.</p>
    </div>
  );
}
```
