#### 1. Клиентская сторона (Browser API)
Работа с `document.cookie` является низкоуровневой и неудобной (работа со строкой).

```JavaScript
// ЧТЕНИЕ COOKIE
// document.cookie возвращает строку вида "name1=value1; name2=value2"
function getCookie(name) {
    const matches = document.cookie.match(new RegExp(
        "(?:^|; )" + name.replace(/([\.$?*|{}\(\)\[\]\\\/\+^])/g, '\\$1') + "=([^;]*)"
    ));
    // Декодируем значение из URL-формата
    return matches ? decodeURIComponent(matches[1]) : undefined;
}

// ЗАПИСЬ COOKIE
// Присвоение не перезаписывает всё, а добавляет/обновляет конкретную запись
function setCookie(name, value, options = {}) {
    options = {
        path: '/',
        // ...другие значения по умолчанию
        ...options
    };

    if (options.expires instanceof Date) {
        options.expires = options.expires.toUTCString();
    }

    let updatedCookie = encodeURIComponent(name) + "=" + encodeURIComponent(value);

    for (let optionKey in options) {
        updatedCookie += "; " + optionKey;
        let optionValue = options[optionKey];
        if (optionValue !== true) {
            updatedCookie += "=" + optionValue;
        }
    }

    document.cookie = updatedCookie;
}

// УДАЛЕНИЕ COOKIE
// Установить Max-Age в -1
function deleteCookie(name) {
    setCookie(name, "", {
        'max-age': -1
    });
}
```

#### 2. Серверная сторона (Node.js + Express)
Фреймворки предоставляют абстракцию для работы с заголовками.

```JavaScript
const express = require('express');
const app = express();

// Установка Cookie (Set-Cookie)
app.get('/login', (req, res) => {
    // res.cookie(name, value, [options])
    res.cookie('authToken', 'secure_token_value_123', {
        httpOnly: true,    // Защита от XSS
        secure: true,      // Только HTTPS
        sameSite: 'strict',// Защита от CSRF
        maxAge: 3600000    // 1 час в миллисекундах
    });
    
    res.send('Cookie установлена');
});

// Чтение Cookie
// Требуется middleware cookie-parser
const cookieParser = require('cookie-parser');
app.use(cookieParser());

app.get('/dashboard', (req, res) => {
    // Доступ через req.cookies
    const token = req.cookies.authToken;
    console.log(token);
    res.send('Проверка токена');
});

// Удаление Cookie
app.get('/logout', (req, res) => {
    // res.clearCookie(name, [options])
    // Важно: опции (path, domain) должны совпадать с теми, что были при создании
    res.clearCookie('authToken', { path: '/' });
    res.send('Вышли из системы');
});
```
