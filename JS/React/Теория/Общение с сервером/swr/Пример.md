import React from "react";

import useSWR from "swr";

function App() {

  return (

    <div>

      <SWRPostsList />

    </div>

  );

}

export default App;

// 1. Функция fetcher (просто обертка над fetch)

const fetcher = async (url) => {

  const response = await fetch(url);

  if (!response.ok) {

    throw new Error("Error loading SWR posts");

  }

  return response.json();

};

const SWRPostsList = () => {

  // 2. Использование useSWR

  // key = URL, fetcher = функция для получения

  const {

    data: posts,

    error,

    isLoading,

  } = useSWR("[https://jsonplaceholder.typicode.com/posts"](https://jsonplaceholder.typicode.com/posts%22 "https://jsonplaceholder.typicode.com/posts%22"), fetcher);

  if (isLoading) return <p>Loading posts SWR...</p>;

  if (error) return <p style={{ color: "red" }}>Error: {error.message}</p>;

  return (

    <div

      style={{

        padding: "20px",

        border: "1px solid #009688",

        marginTop: "20px",

      }}

    >

      <h2>List of posts (SWR)</h2>

      {posts.slice(0, 5).map((post) => (

        <article

          key={post.id}

          style={{ marginBottom: "10px", borderBottom: "1px dotted #e0f2f1" }}

        >

          <h4>{post.title}</h4>

        </article>

      ))}

    </div>

  );

};
