```JSX
import React, { useEffect, useState } from "react";

function App() {
  return (
    <>
      <MouseTracker />
    </>
  );
}

function MouseTracker() {
  const [position, setPosition] = useState({ x: 0, y: 0 });

  // Побочный эффект: подписываемся на событие движения мыши
  useEffect(() => {
    function handleMove(event) {
      setPosition({ x: event.clientX, y: event.clientY });
    }

    // Добавляем обработчик события — это взаимодействие с внешним миром
    window.addEventListener("mousemove", handleMove);
    // Возвращаем функцию очистки — важная часть side effects
    return () => {
      window.removeEventListener("mousemove", handleMove);
    };
  }, []); // Пустой массив: подписка создаётся один раз

  return (
    <div style={{ padding: 20 }}>
      <h1>Mouse Tracker</h1>
      <p>Move your mouse around!</p>
      <pre
        style={{
          background: "#f3f3f3",
          padding: 12,
          borderRadius: 12,
          display: "inline-block",
          marginTop: 12,
        }}
      >
        {JSON.stringify(position, null, 2)}
      </pre>
    </div>
  );
}

export default App;
```
