#### Экспорт компонентов
Требует фигурных скобок в input
```jsx
// MyComponents.jsx
export const Header = () => {
    return <h1>Hello, world!</h1>;
};

export const Footer = () => {
    return <footer>Goodbye, world!</footer>;
};

-----

// App.jsx

import { Header } from "./components/MyComponents.jsx"

import { Footer } from "./components/MyComponents.jsx"

function App() {
  return (
    <div>
      <Header />
      <Footer />
    </div>
  )
}
```

#### default export
Отличается тем, что не требует фигурных скобок при импорте и охотнее подставляется при вводе

Тут экспортируется только `Header`
```jsx
// MyComponents.jsx
const Header = () => {
    return <h1>Hello, world!</h1>;
};

const Footer = () => {
    return <footer>Goodbye, world!</footer>;
};

export default Header;

---

// App.jsx
import Header from "./components/MyComponents.jsx"

function App() {
  return (
    <div>
      <Header />
    </div>
  )
}

export default App;
```


#### Смешанный export
Можно один элемент экспортировать по умолчанию 
```jsx
// MyComponents.jsx
const Header = () => {
    return <h1>Hello, world!</h1>;
};

const Footer = () => {
    return <footer>Goodbye, world!</footer>;
};

export default Header;
export {Footer};

---

// App.jsx
import Header from "./components/MyComponents.jsx"
import { Footer } from "./components/MyComponents.jsx"

function App() {
  return (
    <div>
      <Header />
      <Footer />
    </div>
  )
}

export default App;
```
