```jsx
const numbers = [1, 2, 3, 4, 5];

  

const users = [

  { id: 101, name: "Anna", role: "Student" },

  { id: 102, name: "Boris", role: "Teacher" },

  { id: 103, name: "Vika", role: "Student" },

];

  

const allProducts = [

  { id: 1, name: "Book", inStock: true, price: 100 },

  { id: 2, name: "Pen", inStock: false, price: 50 },

  { id: 3, name: "Notebook", inStock: true, price: 300 },

];

  

const reduce = numbers.reduce((acc, current) => acc + current);

console.log(reduce);

  

class App extends Component

{

  render()

  {

    return(

      <div>

        <ul>

          {users.map((user) => (

            <li key={user.id}>

              {user.name} - {user.role}

            </li>))}

        </ul>

        <ul>

          {allProducts

            .filter((p) => p.inStock)

            .map((p) => (

              <li key={p.id}>

                {p.name}

              </li>

            ))}

        </ul>

        <div>

          {allProducts

            .filter((p) => p.inStock)

            .reduce((sum, current) => {

            }, 0)}

        </div>

      </div>

    );

  }

}
```
