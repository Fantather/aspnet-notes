```JSX
import { useState } from "react";

function App() {
  return (
    <div>
      <WildlandsSelector />
    </div>
  );
}

export default App;

function WildlandsSelector() {
  const [selectedWildlands, setSelectedWildlands] = useState([]);
  const handleMultiSelectChange = (e) => {
    const options = e.target.selectedOptions;
    const values = Array.from(options).map((option) => option.value);
    setSelectedWildlands(values);
    console.log("Selected values:", values);
  };

  return (
    <div>
      <h2>Select multiple Wildlands:</h2>
      <select multiple onChange={handleMultiSelectChange}>
        <option value="mountains">Mountains</option>
        <option value="forest">Forest</option>
        <option value="desert">Desert</option>
        <option value="tundra">Tundra</option>
        <option value="savannah">Savannah</option>
      </select>

      <hr />
      
      <h3>✨ Current Selection State:</h3>
      {selectedWildlands.length === 0 ? (
        <p>Nothing selected</p>
      ) : (
        <p>
          Selected Wildlands: <strong>{selectedWildlands.join(", ")}</strong>
        </p>
      )}
    </div>
  );
}
```
