Если обработчиков много, используем эту микро-фабрику
```JSX
function App() {
  function handleSelect(id) {
    console.log(`Selected item - ${id}`);
  }

  const handleSelectFactory = (id) => () => handleSelect(id);

  return (
    <div>
      <button onClick={handleSelectFactory(1)}>Select</button>
    </div>
  );
}

export default App;
```
