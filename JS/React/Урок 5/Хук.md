Это самая обычная функция

UseState используется для хранения локального состояния функциональных компонентов
Принимает начальное значение состояния и возвращает массив из двух компонентов

`count` это переменная хранящая текущее значение
`setCount` это функция, которая позволяет изменять хранимое значение

```JSX
import { useState } from "react";


export default function App()
{
  return <div>
    <Counter />
  </div>;
}


function Counter()
{
  const [count, setCount] = useState(0);
  return <div>
    <p>Count: {count}</p>
    <button onClick={()=>{setCount(count + 1)}}>Увеличить</button>
  </div>;
}
```


```JSX
import { useState } from 'react';

export default function Counter() {
  const [count, setCount] = useState(0);
  const [name, setName] = useState('Default name');   // string
  const [isOpen, setIsOpen] = useState(false); // boolean
  const names = ['Alice', 'Bob', 'Charlie', 'Diana', 'Eve'];
  
  function changeName() {
    setName(prev => {
      let newName;
      do {
        newName = names[Math.floor(Math.random() * names.length)];
      } while (newName === prev); // чтобы имя отличалось от текущего

      return newName;
    });
  }

  function changeState() {
    setIsOpen(prev => !prev); // переключаем булевое состояние
  }

  return (
    <div>
      <p>Score: {count}</p>
      <button onClick={() => setCount(prev => prev + 1)}>Increment</button>
      <p>Name: {name}</p>
      <button onClick={changeName}>Change name</button>
      <p>Is open: {isOpen.toString()}</p>
      <button onClick={changeState}>Toggle state</button>
    </div>
  );
}
```
