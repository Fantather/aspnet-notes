```JSX
import React, { useState, createContext, useContext } from "react";

// 1. Create UserContext

const UserContext = createContext(null);

// 2. Header component that consumes context

const Header = () => {

  const user = useContext(UserContext);

  return (

    <header className="bg-blue-600 text-white p-4 shadow-lg rounded-t-xl">

      <h1 className="text-2xl font-bold">

        {user ? `Hello, ${user.name}!` : "Loading..."}

      </h1>

      {user && (

        <p className="text-sm opacity-90">

          Your role: <span className="font-semibold">{user.role}</span>

        </p>

      )}

    </header>

  );

};

// 3. Deeply nested component

const UserInfo = () => {

  const user = useContext(UserContext);

  if (!user) {

    return <p className="text-gray-500">User data is not available.</p>;

  }

  return (

    <div className="mt-4 p-4 border border-blue-200 bg-blue-50 rounded-lg">

      <h3 className="text-lg font-semibold text-blue-700">

        Nested Component: UserInfo

      </h3>

      <p className="mt-1 text-gray-700">

        Your unique ID:

        <span className="font-mono bg-blue-100 px-2 py-0.5 rounded text-sm">

          {user.id}

        </span>

      </p>

    </div>

  );

};

// 4. Page container

const Page = () => {

  return (

    <main className="p-6">

      <h2 className="text-xl font-bold mb-3 text-gray-800">Page Content</h2>

      <p className="text-gray-600">

        This component does not receive user props, but it contains the UserInfo

        component which consumes the context.

      </p>

      <UserInfo />

    </main>

  );

};

// 5. Main App component

export default function App() {

  const [user] = useState({

    id: 42,

    name: "John Smith",

    role: "Admin",

  });

  return (

    <div className="min-h-screen bg-gray-100 p-8 flex items-start justify-center">

      <div className="w-full max-w-2xl bg-white shadow-2xl rounded-xl overflow-hidden">

        <UserContext.Provider value={user}>

          <Header />

          <Page />

        </UserContext.Provider>

      </div>

    </div>

  );

}
```
