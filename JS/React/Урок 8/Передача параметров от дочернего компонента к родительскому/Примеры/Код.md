
```JSX
import { useState } from "react";

function App() {

  return (
    <div>
      <SearchBox />
    </div>
  );
}

export default App;

// Дочерний компонент передает введенное значение
function UserInput(props) {
  const handleChange = (event) => {
    // Вызываем функцию родителя и передаем ей значение поля ввода
    props.onChange(event.target.value);
  };

  return <input type="text" onChange={handleChange} placeholder="Enter text" />;

}

function SearchBox() {
  const [searchTerm, setSearchTerm] = useState("");
  
  // Функция, которая принимает данные (value) от дочернего компонента
  const handleInputChange = (newValue) => {
    setSearchTerm(newValue);
  };

  return (
    <div>
      <p>You are searching for: {searchTerm || "nothing"}</p>
      {/* Передаем функцию, которая ожидает один аргумент */}
      <UserInput onChange={handleInputChange} />
    </div>
  );
}
```
