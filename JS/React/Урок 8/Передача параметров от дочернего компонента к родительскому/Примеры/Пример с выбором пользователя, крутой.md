```JSX
import { useState } from "react";

export default function App() {
  return <UsersPage />;
}

/* =================== USERS PAGE (PARENT) =================== */

function UsersPage() {
  const [users, setUsers] = useState([
    { id: 1, name: "Alice", role: "Admin", active: true },
    { id: 2, name: "Bob", role: "User", active: true },
    { id: 3, name: "Charlie", role: "User", active: false },
  ]);
  const [search, setSearch] = useState("");
  const [selectedUserId, setSelectedUserId] = useState(null);

  const selectedUser = users.find((u) => u.id === selectedUserId);
  const filteredUsers = users.filter((u) =>
    u.name.toLowerCase().includes(search.toLowerCase())
  );

  const toggleUserStatus = () => {
    setUsers((prev) =>
      prev.map((u) =>
        u.id === selectedUserId ? { ...u, active: !u.active } : u
      )
    );
  };

  return (
    <div>
      <h1>User Management</h1>
      <UserSearch value={search} onChange={setSearch} />
      <UserList
        users={filteredUsers}
        selectedUserId={selectedUserId}
        onSelect={setSelectedUserId}
      />
      <UserDetails user={selectedUser} />
      <UserActions user={selectedUser} onToggleStatus={toggleUserStatus} />
    </div>
  );
}

/* =================== SEARCH =================== */

function UserSearch({ value, onChange }) {
  return (
    <input
      placeholder="Search users..."
      value={value}
      onChange={(e) => onChange(e.target.value)}
    />
  );
}

/* =================== LIST =================== */

function UserList({ users, selectedUserId, onSelect }) {
  return (
    <ul>
      {users.map((user) => (
        <UserListItem
          key={user.id}
          user={user}
          selected={user.id === selectedUserId}
          onClick={() => onSelect(user.id)}
        />
      ))}
    </ul>
  );
}

function UserListItem({ user, selected, onClick }) {
  return (
    <li
      onClick={onClick}
      style={{
        cursor: "pointer",
        fontWeight: selected ? "bold" : "normal",
      }}
    >
      {user.name}
    </li>
  );
}

/* =================== DETAILS =================== */

function UserDetails({ user }) {
  if (!user) {
    return <p>No user selected</p>;
  }

  return (
    <div>
      <h3>User Details</h3>
      <p>Name: {user.name}</p>
      <p>Role: {user.role}</p>
      <p>Status: {user.active ? "Active" : "Blocked"}</p>
    </div>
  );
}

/* =================== ACTIONS =================== */

function UserActions({ user, onToggleStatus }) {
  if (!user) return null;
  return (
    <button onClick={onToggleStatus}>
      {user.active ? "Block user" : "Unblock user"}
    </button>
  );
}
```
