```JSX
import React, { useState } from "react";

function App() {

  return (

    <div>

      <Parent />

    </div>

  );

}

export default App;

// ---------------------- PARENT ----------------------

function Parent() {

  // 1. Состояние "поднято" до общего предка

  const [selected, setSelected] = useState(null);

  return (

    <>

      {/* Передаем вниз состояние и функцию изменения */}

      <List items={[1, 2, 3]} selected={selected} onSelect={setSelected} />

      {/* Отображаем детали выбранного элемента */}

      <Details id={selected} />

    </>

  );

}

// ---------------------- LIST ----------------------

function List({ items, selected, onSelect }) {

  const handleClick = (itemId) => {

    onSelect(itemId); // поднимаем данные наверх

  };

  return (

    <div>

      {items.map((item) => (

        <button

          key={item}

          className={selected === item ? "selected" : ""}

          onClick={() => handleClick(item)}

        >

          Element {item}

        </button>

      ))}

    </div>

  );

}

// ---------------------- DETAILS ----------------------

function Details({ id }) {

  return (

    <p>

      Details for ID: <strong>{id === null ? "Nothing selected" : id}</strong>

    </p>

  );

}
```
